https://www.acmicpc.net/problem/19532

 

단순히 -999에서 999까지 모두 x,y에 넣어서

두 식이 모두 참이면 답이다.

 

#include <iostream>
using namespace std;

int a,b,c,d,e,f;

int main()
{
    cin >> a >> b >> c >> d >> e >> f;

    for(int x = -999; x <= 999; x++)
    {
        for(int y = -999; y <= 999; y++)
        {
            if((a*x + b*y == c) && (d*x + e*y == f))
            {
                cout << x << ' ' << y;
                return 0;
            }
        }
    }
}
