2178. 미로탐색 과 같지만 다른점은 익은 토마토가 여러개 있을수 있어, 처음에 큐에 1개 이상의 값이 들어간다.

 

처음에 입력 받을때, 익은 토마토라면 미리 큐에 다 넣어준다.

 for(int i = 0; i < H; i++) {
        for(int j = 0; j < W; j++) {
            if(tomato[i][j] == 1) { // 이미 익은 토마토
                q.push(make_pair(i,j)); // 큐에 다 미리 넣어놈
                mark[i][j] = 0; // 이미 익었으므로 익는데 걸리는 시간 : 0일
            }
        }
    }

 

 

#include <iostream>
using namespace std;
#include <utility>
#include <queue>

int tomato[1001][1001];
int mark[1001][1001];
int H,W;
int index = 0;
queue<pair<int,int>> q;

int dc[4] = {0, 1, 0, -1};
int dr[4] = {-1, 0, 1, 0};

void bfs() {

    while(!q.empty()) {
        int nr = q.front().first;
        int nc = q.front().second;
        q.pop();
        if(mark[nr][nc] > index) index++;

        for(int i = 0; i < 4; i++) {
            int nnr = nr + dr[i];
            int nnc = nc + dc[i];

            if(nnr < 0 || nnr >= H || nnc < 0 || nnc >= W) continue;

            if(tomato[nnr][nnc] == 0 && mark[nnr][nnc] == -1) {
                q.push(make_pair(nnr,nnc));
                mark[nnr][nnc] = index + 1;
            }

        }
    }

}

int analyze() {
    int res = 0;

    for(int i = 0; i < H; i++) {
        for(int j = 0; j < W; j++) {
            if(tomato[i][j] == 0 && mark[i][j] == -1) return -1;

            res = max(res, mark[i][j]);
        }
    }
    return res;
}

int main() {
    cin >> W >> H;

    for(int i = 0; i < H; i++) {
        for(int j = 0; j < W; j++) {
            cin >> tomato[i][j];
            mark[i][j] = -1;
        }
    }

    for(int i = 0; i < H; i++) {
        for(int j = 0; j < W; j++) {
            if(tomato[i][j] == 1) {
                q.push(make_pair(i,j));
                mark[i][j] = 0;
            }
        }
    }

    bfs();

    cout << analyze() << endl;
}
