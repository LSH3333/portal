4963. 섬의 개수

PS

2020-08-19 18:13:08

https://www.acmicpc.net/problem/4963

바로 전에 푼 2667. 단지번호붙이기(https://www.acmicpc.net/problem/2667)와 거의 똑같아서 금방 풀 수 있었다.

단지번호붙이기 에서는 상,하,좌,우로 붙어있는 집들이 단지 하나였다면,

섬의 개수 에서는 동,서,남,북,북동,북서,남동,남서로 붙어있는 섬들이 하나의 섬이다. 

따라서 단지번호붙이기 dfs 에서 상,하,좌,우에다가 북동,북서,남동,남서 만 추가해주면 된다.

 

코드

import java.util.*;

public class Main {
    static int[][] arr; // 지도 저장
    static int[][] visit; // 방문여부
    static int w=1, h=1;

    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);

        while(true) {
            w = sc.nextInt(); // 너비 w
            h = sc.nextInt(); // 높이 h
            if(w == 0 && h == 0) break;

            int island = 0;

            arr = new int[h][w];
            visit = new int[h][w];

            // 지도 저장
            for (int i = 0; i < h; i++) {
                for (int j = 0; j < w; j++) {
                    arr[i][j] = sc.nextInt();
                }
            }

            for (int i = 0; i < h; i++) {
                for (int j = 0; j < w; j++) {
                    // 방문 하지 않았고 땅이 존재
                    if (visit[i][j] == 0 && arr[i][j] == 1) {
                        island++;
                        dfs(i, j);
                    }
                }
            }

            System.out.println(island);
        }

    }

    static void dfs(int a, int b) { // a:h, b:w
        visit[a][b] = 1; // 방문 체크

        // 동,서,남,북,대각선 이동
        if(b+1 < w && arr[a][b+1] == 1 && visit[a][b+1] == 0) // 동
            dfs(a, b+1);
        if(b-1 >= 0 && arr[a][b-1] == 1 && visit[a][b-1] == 0) // 서
            dfs(a, b-1);
        if(a-1 >= 0 && arr[a-1][b] == 1 && visit[a-1][b] == 0) // 북
            dfs(a-1, b);
        if(a+1 < h && arr[a+1][b] == 1 && visit[a+1][b] == 0) // 남
            dfs(a+1, b);
        if(b-1 >= 0 && a-1 >= 0 && arr[a-1][b-1] == 1 && visit[a-1][b-1] == 0) // 북서
            dfs(a-1, b-1);
        if(b+1 < w && a-1 >= 0 && arr[a-1][b+1] == 1 && visit[a-1][b+1] == 0) // 북동
            dfs(a-1, b+1);
        if(a+1 < h && b-1 >= 0 && arr[a+1][b-1] == 1 && visit[a+1][b-1] == 0) // 남서
            dfs(a+1, b-1);
        if(a+1 < h && b+1 < w && arr[a+1][b+1] == 1 && visit[a+1][b+1] == 0) // 남동
            dfs(a+1, b+1);

    }
}
