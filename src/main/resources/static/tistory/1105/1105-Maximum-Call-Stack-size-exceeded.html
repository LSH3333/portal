Maximum Call Stack size exceeded

Web

2023-04-16 10:41:31

 

var byteArray = new Uint8Array(event.target.result);
var binaryString = String.fromCharCode.apply(null, byteArray);
var base64String = btoa(binaryString);

byte array 를 binary String 으로 변환하는 과정에서 Maximum Call Stack size exceeded 에러가 발생.

 

원인:

웹 브라우저는 너무 많은 argument 를 처리할수 없기 때문에 발생.

apply() 를 쓰지 않고 일일히 fromCharCode로 변환후 binary 문자열에 추가해주면 된다. 

 


function byteArrayToBase64String(bytes) {
let binary = '';
for(let i = 0; i < bytes.length; i++) {
binary += String.fromCharCode(bytes[i]);
}
return btoa(binary);
}
