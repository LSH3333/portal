top-down 재귀

#include <iostream>
using namespace std;

int d[21];

/// top-down
int pibo(int n) {
    /// base
    if(n <= 1) return n;
    if(d[n] > 0) return d[n]; /// memoization
    return d[n] = pibo(n-1) + pibo(n-2);
}

int main() {
    int n;
    cin >> n;

    cout << pibo(n);

}

 

bottom-up 재귀

#include <iostream>
using namespace std;

int d[21];

/// bottom-up
int pibo(int n) {
    /// base
    d[0] = 0;
    d[1] = 1;

    for(int i = 2; i <= n; i++) {
        d[i] = d[i-1] + d[i-2];
    }
    return d[n];
}

int main() {
    int n;
    cin >> n;

    cout << pibo(n);
}
