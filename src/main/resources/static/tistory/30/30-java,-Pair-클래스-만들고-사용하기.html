java는 pair클래스가 없는 것 같다. 

직접 만들고 사용하기.

 

import java.util.*;

class Pair { // Queue에 넣을 데이터 클래스
    int x, y;
    public Pair(int _x, int _y) {
        x = _x;
        y = _y;
    }
}

public class Sub {
    public static void main(String args[]) {
        Queue<Pair> q = new LinkedList<Pair>();

        q.add(new Pair(1, 2));
        Pair _p = q.remove();
        System.out.println(_p.x + " " + _p.y);
    }
}
