1. 1번부터 n번까지의 아이스크림 중 3가지의 조합을 골라서

   그 아이스크림 조합에 맛없는 두개의 아이스크림이 포함되어있지 않으면 cnt를 1 증가시킨다.

 

2. cnt를 출력한다 

 

금방 풀수 있을줄 알았는데 시간초과 때문에 애먹었다.

맛없는 조합을 입력을 받을때 2차원 배열로 입력을 받아서 해결.

 

#include <iostream>
#include <vector>
using namespace std;

int n,m;
int ans = 0;
bool mark[201];
bool bad[201][201];
vector<int> ice;


// 현재 벡터 ice에 존재하는 아이스크림 조합이 맛있는 조합인가?
// 맛있는 조합이면 true 리턴
// 맛없는 조합이면 false 리턴
bool IsPossible()
{
    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 3; j++)
        {
            if(i == j) continue;

            if(bad[ice[i]][ice[j]]) return false;
        }
    }
    return true;
}

void dfs(int idx, int depth)
{
    if(depth == 3)
    {
        if(IsPossible())
            ans++;

        return;
    }

    for(int i = idx; i <= n; i++)
    {
        if(mark[i]) continue;

        ice.push_back(i);
        mark[i] = true;

        dfs(i, depth+1);

        mark[i] = false;
        ice.pop_back();
    }
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n >> m;
    for(int i = 0; i < m; i++)
    {
        int a, b;
        cin >> a >> b;
        // 맛없는 조합을 2차원 배열로 입력받는다
        bad[a][b] = true;
        bad[b][a] = true;
    }

    // 아이스크림은 1번부터 존재
    dfs(1,0);

    cout << ans;
}
