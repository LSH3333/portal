동전들의 액수를 내림차순으로 배열에 저장해 놓는다. 

내야할돈의 액수를 동전의 액수만큼 배면서 카운트를 증가시킨다.

 

#include <iostream>
#include <vector>
using namespace std;

int n;
int coin[6] = {500,100,50,10,5,1};

int main()
{
    cin >> n;

    int pay = 1000-n;
    int cnt = 0;
    for(int i = 0; i < 6; i++)
    {
        while(pay >= coin[i])
        {
            pay -= coin[i];
            cnt++;
        }
    }

    cout << cnt;
}
