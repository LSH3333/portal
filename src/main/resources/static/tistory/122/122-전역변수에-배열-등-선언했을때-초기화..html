코테 문제중 여러개의 테스트케이스로 이루어진 경우에,

전역변수에 배열 등 데이터를 선언했다면 초기화를 꼭 해주자.

 

#include <iostream>
using namespace std;
#include <cstring>

bool mark[10];
vector<int> v;

int main() {
	int t; // test case
    cin >> t;
    
    while(t--) {
    	// 테스트 케이스가 다시 시작될때 꼭 초기화!
        memset(mark, false, sizeof(mark));
        v.clear(); // v.clear()는 메모리는 초기화 되지 않는다.
        
        
        /*
        코드들 ...
        */
    }
}
