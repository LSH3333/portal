주어진 숫자를 오름차순 정렬후 첫요소와 두번째요소를 더한 값을 첫요소와 두번째요소와 대체한다.

이것을 m번 반복한후 전체 배열 요소들의 합을 출력한다.

 

처음에 틀렸다고 나왔는데 알고보니 foreach문에서 모든 요소를 더할때 long long이나 auto가 아닌 int형으로 x를 선언해서 틀렸다.

 

 

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int n, m;
long long ans = 0;
vector<long long> v;


int main()
{
    cin >> n >> m;
    for(int i = 0; i < n; i++)
    {
        long long num;
        cin >> num;
        v.push_back(num);
    }

    for(int i = 0; i < m; i++)
    {
        sort(v.begin(), v.end());
        long long sum = v[0] + v[1];
        v[0] = sum;
        v[1] = sum;
    }


    for(auto x : v)
    {
        ans += x;
    }

    cout << ans;



}
