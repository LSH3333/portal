이번에는 재귀를 이용해서 풀어본다.

 

 

#include <iostream>
using namespace std;

int a[11];
int num;
int cnt = 0;

// a[0]부터 a[idx]까지 원소들의 합 리턴
int sums(int idx) {
    int sum = 0;
    for(int i = 0; i <= idx; i++) {
        sum += a[i];
    }
    return sum;
}

void cal(int idx) {
    // base check
    if(idx > num) return; // 배열의길이는 최대 num 이상이 될일이없음

    for(int i = 1; i <= 3; i++) {
        a[idx] = i;
        if(sums(idx) == num) cnt++;
        cal(idx+1); // 재귀호출
    }
}

int main() {
    int t;
    cin >> t;

    while(t--) {
        cin >> num;
        cnt = 0;
        cal(0);
        cout << cnt << '\n';
    }


}

 

처음에 재귀의 base check를 idx > 10 이면 리턴하도록 했는데 생각해보니 어처피 배열의 길이가 내가 구하는 숫자인 num보다 길어지면 무조건 num값이 넘으므로 (num = 4인데 배열길이가 5면 무조건 5이상값이됨) 

num보다 크면 리턴하도록 했더니 실행시간이 많이 짧아졌다.
