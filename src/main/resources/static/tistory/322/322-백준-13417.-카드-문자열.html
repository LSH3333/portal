문자를 리스트에 넣으면서 다음 문자가 리스트의 front()의 문자보다 아스키 코드상 작은값이면 push_front()해주고,

큰값이면 push_back() 해주면 된다.

 

#include <iostream>
#include <vector>
#include <list>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    int tc;
    cin >> tc;

    while(tc--)
    {
        int n;
        vector<char> v;
        list<char> l;

        // inputs
        cin >> n;
        v.resize(n);
        for(int i = 0; i < n; i++)
            cin >> v[i];

        l.push_back(v[0]);
        for(int i = 1; i < n; i++)
        {
            // 리스트의 front에 저장된 문자보다 빠른 알파벳이면
            if(v[i] <= l.front())
            {
                l.push_front(v[i]);
            }
            else
            {
                l.push_back(v[i]);
            }
        }

        for(auto x : l)
            cout << x;
        cout << '\n';
    }

}
