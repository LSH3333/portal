재귀에 대해 뭔가 모호하게 아는것같아서 재귀문제들을 좀 풀어보려고 한다!

 

이 문제는 가장 기본적인 재귀함수 문제.

재귀함수에서 탈출조건만 잘 만족시켜주면 쉽게 풀린다.

 

그런데 정답을 냈는데 계속 1퍼센트에서 틀려서 보니까 

앞에 나오는 문자열이 "-"이 아닌 "_" 이었다.. 햇갈려요..

 

#include <iostream>
using namespace std;
int n;

void print(int num) {
    for(int i = 0; i < num*4; i++) cout << "_";
}

void recur(int num) {

    print(num);
    cout << "\"재귀함수가 뭔가요?\"" << '\n';

    if(num == n) {
        print(num);
        cout << "\"재귀함수는 자기 자신을 호출하는 함수라네\"" << '\n';
        print(num);
        cout << "라고 답변하였지.";
        cout << '\n';
        return;
    }

    print(num);
    cout << "\"잘 들어보게. 옛날옛날 한 산 꼭대기에 이세상 모든 지식을 통달한 선인이 있었어." << '\n';
    print(num);
    cout << "마을 사람들은 모두 그 선인에게 수많은 질문을 했고, 모두 지혜롭게 대답해 주었지." << '\n';
    print(num);
    cout << "그의 답은 대부분 옳았다고 하네. 그런데 어느 날, 그 선인에게 한 선비가 찾아와서 물었어.\"" << '\n';

    recur(num+1);

    print(num);
    cout << "라고 답변하였지.";
    if(num != 0) cout << '\n';


}

int main() {
    cin >> n;
    cout << "어느 한 컴퓨터공학과 학생이 유명한 교수님을 찾아가 물었다." << '\n';
    recur(0);
}
