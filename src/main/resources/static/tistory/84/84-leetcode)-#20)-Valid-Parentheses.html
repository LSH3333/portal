ool isValid(string s) {
    stack<char> st;
    int len = s.size();

    st.push(s[0]);
    for(int i = 1; i < len; i++) {
        // if it's closing
        if(s[i] == ')') {
            if(st.empty() || st.top() != '(') return false;
            if(st.top() == '(') st.pop();
        }
        else if(s[i] == ']') {
            if(st.empty() || st.top() != '[') return false;
            if(st.top() == '[') st.pop();
        }
        else if(s[i] == '}') {
            if(st.empty() || st.top() != '{') return false;
            if(st.top() == '{') st.pop();
        }
        else { // opening
            st.push(s[i]);
        }
    }

    if(st.empty()) return true;
    else return false;

}
