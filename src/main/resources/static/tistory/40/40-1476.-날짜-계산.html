#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    /// 1 <= E <= 15, 1 <= S <= 28, 1 <= M <= 19
    int E,S,M;
    cin >> E >> S >> M;

    int res = 0;
    int e = 0, s = 0, m = 0;
    while(true) {
        res++;
        e++; s++; m++;

        //cout << e << " " << s << " "<< m << endl;
        if(e == E && s == S && m == M) break;

        if(e == 15) e = 0;
        if(s == 28) s = 0;
        if(m == 19) m = 0;
    }

    cout << res;
}
