삼각형의 성립조건은 a+b > c (c가 가장 긴 변)

 

빨대 길이를 내림차순으로 정렬하고

i, i+1, i+2 세 변을 비교해서 i(가장긴변) < (i+1) + (i+2)가 성립한다면 만들수있는 세변의 길이가 가장긴 삼각형이다.

 

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int n;
vector<int> v;

int main()
{
    cin >> n;
    v.resize(n);
    for(int i = 0; i < n; i++)
        cin >> v[i];

    // 내림차순 정렬
    sort(v.begin(), v.end(), greater<>());

    for(int i = 0; i < n-2; i++)
    {
        // 삼각형 성립 조건
        if(v[i] < v[i+1] + v[i+2])
        {
            cout << v[i]+v[i+1]+v[i+2];
            return 0;
        }
    }

    // 삼각형 만들수 없음
    cout << -1;
}
