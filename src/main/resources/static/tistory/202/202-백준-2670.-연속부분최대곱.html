실버4로 책정된 문제치고 너무 쉽다 싶었는데 dp로 풀수 있어서 그런것같다.

나는 브루트포스 알고리즘으로 풀었다.

 

인덱스 i부터 i+1, i+2 .. i+n-1 까지 곱하면서 최댓값을 계속 갱신하고 i값을 1씩 증가시키면 된다.

 

#include <iostream>
#include <algorithm>
using namespace std;

int n;
double ans = 0;
double nums[10001];

// 최댓값 갱신
void Calculate(double num)
{
    ans = max(ans, num);
}

int main()
{
    cin >> n;
    for(int i = 0; i < n; i++)
    {
        cin >> nums[i];
    }

    for(int i = 0; i < n; i++)
    {
        double num = nums[i];
        Calculate(num);
        for(int j = i+1; j < n; j++)
        {
            num *= nums[j];
            Calculate(num);
        }
    }

    printf("%.3f", ans);
}
