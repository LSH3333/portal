그냥 dfs, bfs 알고리즘을 사용하면 되는데 주의할점은 조건중에,

방문할 수 있는 정점이 여러 개인 경우에는 정점 번호가 작은 것을 먼저 방문하고..

이것때문에 그래프를 입력받고 sort를 해주고 dfs, bfs 계산을 해야한다.

 

#include <iostream>
#include <vector>
#include <stack>
#include <queue>
using namespace std;
#include <algorithm>

vector<int> edge[10001];
bool mark_dfs[1001];
bool mark_bfs[1001];

void dfs(int n) {
    if(mark_dfs[n]) return;
    mark_dfs[n] = true;
    cout << n << ' ';

    for(int x : edge[n]) dfs(x);
}

void bfs(int n) {
    queue<int> q;
    mark_bfs[n] = true;
    q.push(n);
    cout << n << ' ';

    while(!q.empty()) {
        int v = q.front();

        for(int x : edge[v]) {
            if(!mark_bfs[x]) {
                q.push(x);
                mark_bfs[x] = true;
                cout << x << ' ';
            }
        }
        q.pop();
    }


}

int main() {
    int N, M, V; // 정점, 간선, 시작정점
    cin >> N >> M >> V;

    // graph input
    for(int i = 0; i < M; i++) {
        int a, b;
        cin >> a >> b;
        edge[a].push_back(b);
        edge[b].push_back(a);
    }

    // sort
    for(int i = 1; i <= N; i++) {
        sort(edge[i].begin(), edge[i].end());
    }
/*
    for(int i = 1; i <= N; i++) {
        for(int x : edge[i]) cout << x << ' ';
        cout << endl;
    }
cout << endl;
*/
    // print
    dfs(V); cout << endl;
    bfs(V);
}
