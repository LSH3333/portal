#include <iostream>
using namespace std;

// num의 각 자릿수를 더한값 리턴
int sums(int num) {
    int sum = 0;
    while(num != 0) {
        sum += num % 10;
        num /= 10;
    }
    return sum;
}

int main() {
    int n;
    int ans=0;
    cin >> n;

    for(int i = 1; i <= n; i++) {
        // i값과 i의 각 자릿수들의 합이 n이라면 i가 생성자
        if(i + sums(i) == n) {
            ans = i;
            break;
        }
    }
    cout << ans;
}

조심할점은 생성자가 없는 자연수도 있고, 그경우 0을 출력해야 하기 때문에

ans값을 0으로 꼭 초기화 해줘야 for문을 돌고 생성자를 못찾았을때 ans를 0으로 출력한다.
